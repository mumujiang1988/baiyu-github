  ```javascript
  // utils/configValidator.js
  export function validateConfig(config) {
    const errors = [];
    
    // 验证必需字段
    if (!config.pageConfig?.moduleCode) {
      errors.push('缺少 pageConfig.moduleCode');
    }
    
    if (!config.tableConfig?.tableName) {
      errors.push('缺少 tableConfig.tableName');
    }
    
    // 验证字典配置
    config.formConfig?.sections?.forEach(section => {
      section.fields.forEach(field => {
        if (field.required && field.dictionary && !config.dictionaryConfig?.[field.dictionary]) {
          errors.push(`字段 ${field.field} 依赖的字典 ${field.dictionary} 未配置`);
        }
      });
    });
    
    return {
      valid: errors.length === 0,
      errors
    };
  }
  
  // 在 loadDatabaseConfig 中使用
  const loadDatabaseConfig = async (moduleCode) => {
    const configContent = await ERPConfigParser.loadFromDatabase(moduleCode);
    
    // ✅ 添加配置验证
    const validationResult = validateConfig(configContent);
    if (!validationResult.valid) {
      console.error('配置验证失败:', validationResult.errors);
      throw new Error(`配置验证失败：${validationResult.errors.join(', ')}`);
    }
    
    currentConfig.value = configContent;
    parser = new ERPConfigParser(configContent);
  }
  ```

  ```javascript
  // utils/xss.js
  export function escapeHtml(str) {
    if (!str) return '';
    const map = {
      '&': '&amp;',
      '<': '&lt;',
      '>': '&gt;',
      '"': '&quot;',
      "'": '&#039;'
    };
    return str.replace(/[&<>"']/g, m => map[m]);
  }
  
  // 在渲染配置内容前转义
  const renderConfig = (config) => {
    return JSON.parse(escapeHtml(JSON.stringify(config)));
  };
  ```

#### 4. SQL 注入防护（架构级保障）
- **问题描述：** 动态查询可能存在 SQL 注入风险
- **影响范围：** 查询引擎
- **架构保障：**
  - ✅ **字段名来自 JSON 配置**（用户无法手动输入）
  - ✅ **表名来自 JSON 配置**（用户无法手动输入）
  - ✅ **运算符来自预定义列表**（eq, ne, gt, lt 等）
  - ✅ **值采用参数化查询**（MyBatis-Plus 自动处理）
  
- **验证方案：**
  ```java
  // DynamicQueryEngine.java
  // ✅ 所有字段名来自配置，用户无法控制
  String field = (String) fieldConfig.get("field");  // 来自 JSON 配置
  
  // ✅ 所有值采用参数化查询
  queryWrapper.eq(field, value.toString());  // MyBatis-Plus 自动参数化
  
  // ❌ 不存在字符串拼接 SQL
  // String sql = "SELECT * FROM " + tableName;  // 不允许这样写
  ```
  
- **补充措施：**
  - 添加日志记录所有生成的查询条件
  - 定期审计配置数据
  - 禁止在配置中使用 SQL 片段
  
- **预期收益：** 通过架构设计杜绝 SQL 注入，而非依赖代码检查
