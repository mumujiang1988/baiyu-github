# 国家模糊查询接口使用文档

## 📋 接口说明

### 后端接口

#### 1. 国家模糊搜索
- **接口路径**: `GET /erp/engine/country/search`
- **描述**: 支持按中文名或英文名模糊搜索国家
- **参数**:
  - `keyword` (可选): 搜索关键词（中文或英文）
  - `limit` (可选): 返回数量限制，默认 20

**请求示例：**
```http
GET /erp/engine/country/search?keyword=中国&limit=10
```

**返回格式：**
```json
{
  "code": 200,
  "msg": "操作成功",
  "data": [
    {
      "id": "156",
      "labelEn": "China",
      "labelZh": "中国",
      "status": 1
    },
    {
      "id": "840",
      "labelEn": "United States",
      "labelZh": "美国",
      "status": 1
    }
  ]
}
```

#### 2. 获取所有国家列表
- **接口路径**: `GET /erp/engine/country/list`
- **描述**: 获取所有启用状态的国家（按中文名排序）
- **参数**: 无

**请求示例：**
```http
GET /erp/engine/country/list
```

**返回格式：**
```json
{
  "code": 200,
  "msg": "操作成功",
  "data": [
    {
      "id": "004",
      "labelEn": "Afghanistan",
      "labelZh": "阿富汗",
      "status": 1
    },
    {
      "id": "008",
      "labelEn": "Albania",
      "labelZh": "阿尔巴尼亚",
      "status": 1
    }
  ]
}
```

## 🎯 前端使用方式

### 方式 1：直接使用 API（推荐）

```vue
<template>
  <div>
    <!-- 国家搜索下拉框 -->
    <el-select 
      v-model="form.country" 
      placeholder="请选择国家"
      filterable
      remote
      :remote-method="searchCountryHandler"
      :loading="loading"
    >
      <el-option
        v-for="item in countryOptions"
        :key="item.id"
        :label="item.labelZh + ' (' + item.labelEn + ')'"
        :value="item.id"
      />
    </el-select>
  </div>
</template>

<script setup>
import { ref, onMounted } from 'vue'
import { searchCountry, getAllCountries } from '@/views/erp/api/country'

const form = ref({
  country: ''
})

const loading = ref(false)
const countryOptions = ref([])

// 搜索处理函数
const searchCountryHandler = async (keyword) => {
  if (!keyword || keyword.trim() === '') {
    // 空关键词时加载所有国家
    await loadAllCountries()
    return
  }
  
  loading.value = true
  try {
    const response = await searchCountry(keyword, 20)
    if (response.code === 200) {
      countryOptions.value = response.data || []
    }
  } catch (error) {
    console.error('搜索国家失败:', error)
  } finally {
    loading.value = false
  }
}

// 加载所有国家
const loadAllCountries = async () => {
  loading.value = true
  try {
    const response = await getAllCountries()
    if (response.code === 200) {
      countryOptions.value = response.data || []
    }
  } catch (error) {
    console.error('加载国家列表失败:', error)
  } finally {
    loading.value = false
  }
}

// 初始化加载
onMounted(async () => {
  await loadAllCountries()
})
</script>
```

## 💡 使用场景

### 场景 1：客户资料 - 国家选择

```vue
<template>
  <el-form-item label="所属国家">
    <el-select 
      v-model="customer.country" 
      placeholder="请选择国家"
      filterable
      remote
      :remote-method="debounce(searchCountry, 300)"
      :loading="countryLoading"
    >
      <el-option
        v-for="item in countryList"
        :key="item.id"
        :label="`${item.labelZh} / ${item.labelEn}`"
        :value="item.id"
      />
    </el-select>
  </el-form-item>
</template>

<script setup>
import { ref } from 'vue'
import { searchCountry } from '@/views/erp/api/country'

const customer = ref({
  country: ''
})

const countryLoading = ref(false)
const countryList = ref([])

// 防抖搜索
const debounce = (fn, delay) => {
  let timer = null
  return function(...args) {
    if (timer) clearTimeout(timer)
    timer = setTimeout(() => {
      fn.apply(this, args)
    }, delay)
  }
}

// 搜索国家
const handleSearchCountry = async (keyword) => {
  countryLoading.value = true
  try {
    const response = await searchCountry(keyword, 20)
    if (response.code === 200) {
      countryList.value = response.data
    }
  } catch (error) {
    console.error('搜索失败:', error)
  } finally {
    countryLoading.value = false
  }
}
</script>
```

### 场景 3：批量导入时的国家验证

```javascript
import { searchCountry } from '@/views/erp/api/country'

/**
 * 验证国家名称
 * @param {string} countryName - 国家名称
 * @returns {Promise<{valid: boolean, countryId: string|null}>}
 */
export async function validateCountry(countryName) {
  try {
    const response = await searchCountry(countryName, 5)
    
    if (response.code === 200 && response.data.length > 0) {
      // 精确匹配
      const exactMatch = response.data.find(
        item => item.labelZh === countryName || item.labelEn === countryName
      )
      
      if (exactMatch) {
        return {
          valid: true,
          countryId: exactMatch.id,
          matchedName: exactMatch.labelZh
        }
      }
      
      // 模糊匹配（第一个结果）
      return {
        valid: true,
        countryId: response.data[0].id,
        matchedName: response.data[0].labelZh
      }
    }
    
    return {
      valid: false,
      countryId: null,
      error: '国家不存在'
    }
  } catch (error) {
    console.error('验证国家失败:', error)
    return {
      valid: false,
      countryId: null,
      error: error.message
    }
  }
}

// 使用示例
const result = await validateCountry('中国')
if (result.valid) {
  console.log('国家 ID:', result.countryId)
} else {
  console.error('验证失败:', result.error)
}
```

```javascript
/**
 * 智能国家搜索（支持拼音、简写）
 */
export async function smartCountrySearch(keyword) {
  // 先尝试标准搜索
  let result = await searchCountry(keyword, 10)
  
  if (result.code === 200 && result.data.length > 0) {
    return result.data
  }
  
  // 如果没有结果，尝试拼音首字母
  const pinyinMap = {
    'zg': '中国',
    'mg': '美国',
    'rg': '日本',
    // ... 更多映射
  }
  
  const full_name = pinyinMap[keyword.toLowerCase()]
  if (full_name) {
    result = await searchCountry(full_name, 10)
    return result.data || []
  }
  
  return []
}
```

### 2. 缓存优化

```javascript
// 使用 localStorage 缓存国家数据
const CACHE_KEY = 'erp_country_cache'
const CACHE_TTL = 24 * 60 * 60 * 1000 // 24 小时

export async function getCachedCountries() {
  const cached = localStorage.getItem(CACHE_KEY)
  
  if (cached) {
    const { data, timestamp } = JSON.parse(cached)
    
    // 检查缓存是否过期
    if (Date.now() - timestamp < CACHE_TTL) {
      console.log('💾 使用缓存的国家数据')
      return data
    }
  }
  
  // 重新加载
  console.log('🌐 从服务器加载国家数据')
  const response = await getAllCountries()
  
  if (response.code === 200) {
    // 更新缓存
    localStorage.setItem(CACHE_KEY, JSON.stringify({
      data: response.data,
      timestamp: Date.now()
    }))
    return response.data
  }
  
  return []
}

// 清除缓存
export function clearCountryCache() {
  localStorage.removeItem(CACHE_KEY)
  console.log('✅ 国家缓存已清除')
}
```

```javascript
const debouncedSearch = debounce(searchCountry, 300)
```

