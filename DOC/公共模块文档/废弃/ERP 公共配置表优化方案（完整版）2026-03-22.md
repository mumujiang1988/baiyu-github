# ERP 公共配置表优化方案（完整版）

> 📅 **创建时间**: 2026-03-22  
> 🎯 **目标**: 构建统一的公共配置表存储 JSON 配置，实现配置化页面驱动  
> 📦 **适用范围**: RuoYi-WMS + Vue 3 + Element Plus  

---

##  优化方案总览

### 核心改进

**原方案风险点** → **新方案解决方案**

| 风险点 | 原方案问题 | 新方案解决方案 |
|--------|----------|--------------|
| **配置复杂度高** | JSON 文件散乱，难以管理 |  统一公共表集中管理 |
| **并发/事务问题** | 下推数据一致性难保证 |  数据库事务 + 乐观锁 |
| **权限控制弱** | 文件系统权限粗糙 |  RBAC 细粒度权限 |
| **版本管理缺失** | 无法追溯历史版本 |  自动版本历史表 |
| **性能瓶颈** | 大 JSON 解析慢 |  Redis 多级缓存 |
| **校验缺失** | 无 Schema 验证 |  JSON Schema + 后端验证 |

### 1.1 公共配置表结构

```sql
-- ============================================
-- ERP 公共配置表 - 核心表
-- ============================================

DROP TABLE IF EXISTS `erp_page_config`;
CREATE TABLE `erp_page_config` (
  `config_id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '主键 ID',
  `module_code` varchar(50) NOT NULL COMMENT '模块编码（如 saleOrder/deliveryOrder）',
  `config_name` varchar(100) NOT NULL COMMENT '配置名称',
  `config_type` varchar(20) NOT NULL DEFAULT 'PAGE' COMMENT '配置类型（PAGE=页面配置/DICT=字典配置/PUSH=下推配置/APPROVAL=审批配置）',
  `config_content` LONGTEXT NOT NULL COMMENT '完整的 JSON 配置内容',
  `version` int(11) NOT NULL DEFAULT 1 COMMENT '版本号（每次更新 +1）',
  `status` char(1) NOT NULL DEFAULT '1' COMMENT '状态（1 正常 0 停用）',
  `is_public` char(1) NOT NULL DEFAULT '0' COMMENT '是否公共配置（1 是 0 否）',
  `parent_config_id` bigint(20) DEFAULT NULL COMMENT '父配置 ID（用于继承）',
  `remark` varchar(500) DEFAULT NULL COMMENT '备注',
  `create_by` varchar(64) DEFAULT '' COMMENT '创建者',
  `create_time` datetime DEFAULT NULL COMMENT '创建时间',
  `update_by` varchar(64) DEFAULT '' COMMENT '更新者',
  `update_time` datetime DEFAULT NULL COMMENT '更新时间',
  PRIMARY KEY (`config_id`),
  UNIQUE KEY `uk_module_type` (`module_code`, `config_type`),
  KEY `idx_status` (`status`),
  KEY `idx_is_public` (`is_public`),
  KEY `idx_parent_config` (`parent_config_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='ERP 公共配置表';
```

### 2.3 Service 服务层（核心）

```java
package com.ruoyi.business.k3.service;

import cn.hutool.core.util.ObjectUtil;
import com.alibaba.fastjson2.JSON;
import com.alibaba.fastjson2.JSONObject;
import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import com.baomidou.mybatisplus.core.toolkit.Wrappers;
import com.ruoyi.business.k3.domain.ErpPageConfig;
import com.ruoyi.business.k3.domain.ErpPageConfigHistory;
import com.ruoyi.business.k3.mapper.ErpPageConfigHistoryMapper;
import com.ruoyi.business.k3.mapper.ErpPageConfigMapper;
import com.ruoyi.common.redis.utils.RedisUtils;
import jakarta.annotation.Resource;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.concurrent.TimeUnit;

/**
 * ERP 公共配置服务
 */
@Slf4j
@RequiredArgsConstructor
@Service
public class ErpPageConfigService {

    private final ErpPageConfigMapper configMapper;
    private final ErpPageConfigHistoryMapper historyMapper;
    @Resource
    private RedisUtils redisUtils;

    private static final String CACHE_KEY_PREFIX = "erp:page:config:";
    private static final long CACHE_TTL_HOURS = 1;

    /**
     * 根据模块编码获取配置（带多级缓存）
     */
    public String getConfigByModule(String moduleCode) {
        // 1. 尝试 Redis 缓存
        String cacheKey = CACHE_KEY_PREFIX + moduleCode;
        String cached = redisUtils.getCacheObject(cacheKey);
        if (ObjectUtil.isNotEmpty(cached)) {
            log.debug("命中 Redis 缓存：{}", moduleCode);
            return cached;
        }

        // 2. 查询数据库
        ErpPageConfig config = configMapper.selectByModuleCode(moduleCode);
        if (config == null || !"1".equals(config.getStatus())) {
            log.warn("未找到配置或配置已停用：{}", moduleCode);
            return null;
        }

        // 3. 写入 Redis 缓存
        redisUtils.setCacheObject(cacheKey, config.getConfigContent(), 
                                   CACHE_TTL_HOURS, TimeUnit.HOURS);
        
        log.info("加载配置成功：{} (v{})", moduleCode, config.getVersion());
        return config.getConfigContent();
    }

    /**
     * 保存配置（带版本管理和事务）
     */
    @Transactional(rollbackFor = Exception.class)
    public void saveConfig(ErpPageConfig config, String changeReason) {
        // 1. 验证 JSON 格式
        try {
            JSON.parseObject(config.getConfigContent());
        } catch (Exception e) {
            throw new IllegalArgumentException("配置必须是有效的 JSON 格式：" + e.getMessage());
        }

        // 2. 新增或更新
        if (config.getConfigId() != null) {
            // 更新：递增版本号
            ErpPageConfig oldConfig = configMapper.selectById(config.getConfigId());
            if (oldConfig == null) {
                throw new RuntimeException("配置不存在");
            }

            // 乐观锁检查
            if (!ObjectUtil.equal(oldConfig.getVersion(), config.getVersion())) {
                throw new RuntimeException("配置已被修改，请刷新后重试");
            }

            // 递增版本号
            config.setVersion(oldConfig.getVersion() + 1);
            configMapper.updateById(config);

            // 记录历史
            saveHistory(oldConfig, config, changeReason, "UPDATE");

            // 清除缓存
            clearCache(oldConfig.getModuleCode());

            log.info("更新配置成功：{} v{} -> v{}", 
                     oldConfig.getModuleCode(), oldConfig.getVersion(), config.getVersion());

        } else {
            // 新增：检查是否已存在
            ErpPageConfig existing = configMapper.selectByModuleCode(
                config.getModuleCode()
            );
            if (existing != null) {
                throw new RuntimeException("配置已存在：" + config.getModuleCode());
            }

            config.setVersion(1);
            configMapper.insert(config);

            // 记录历史
            saveHistory(null, config, changeReason, "ADD");

            log.info("新增配置成功：{}", config.getModuleCode());
        }
    }

    /**
     * 回滚到指定版本
     */
    @Transactional(rollbackFor = Exception.class)
    public void rollbackToVersion(Long configId, Integer targetVersion, String reason) {
        // 1. 查找目标版本历史
        ErpPageConfigHistory history = historyMapper.selectOne(
            new LambdaQueryWrapper<ErpPageConfigHistory>()
                .eq(ErpPageConfigHistory::getConfigId, configId)
                .eq(ErpPageConfigHistory::getVersion, targetVersion)
        );

        if (history == null) {
            throw new RuntimeException("指定的版本不存在：" + targetVersion);
        }

        // 2. 更新当前配置
        ErpPageConfig current = configMapper.selectById(configId);
        current.setConfigContent(history.getConfigContent());
        current.setVersion(current.getVersion() + 1);
        configMapper.updateById(current);

        // 3. 记录回滚历史
        ErpPageConfigHistory rollbackHistory = new ErpPageConfigHistory();
        rollbackHistory.setConfigId(configId);
        rollbackHistory.setModuleCode(current.getModuleCode());
        rollbackHistory.setConfigType(current.getConfigType());
        rollbackHistory.setVersion(current.getVersion());
        rollbackHistory.setConfigContent(history.getConfigContent());
        rollbackHistory.setChangeReason(reason);
        rollbackHistory.setChangeType("ROLLBACK");
        historyMapper.insert(rollbackHistory);

        // 4. 清除缓存
        clearCache(current.getModuleCode());

        log.info("回滚配置成功：{} v{} -> rollback to v{}", 
                 current.getModuleCode(), current.getVersion(), targetVersion);
    }

    /**
     * 获取配置历史列表
     */
    public List<ErpPageConfigHistory> getConfigHistory(Long configId) {
        return historyMapper.selectList(
            new LambdaQueryWrapper<ErpPageConfigHistory>()
                .eq(ErpPageConfigHistory::getConfigId, configId)
                .orderByDesc(ErpPageConfigHistory::getVersion)
        );
    }

    /**
     * 清除缓存
     */
    private void clearCache(String moduleCode) {
        String cacheKey = CACHE_KEY_PREFIX + moduleCode;
        redisUtils.deleteObject(cacheKey);
        log.debug("清除缓存：{}", cacheKey);
    }

    /**
     * 保存历史记录
     */
    private void saveHistory(ErpPageConfig oldConfig, ErpPageConfig newConfig, 
                            String reason, String changeType) {
        ErpPageConfigHistory history = new ErpPageConfigHistory();
        history.setConfigId(newConfig.getConfigId());
        history.setModuleCode(newConfig.getModuleCode());
        history.setConfigType(newConfig.getConfigType());
        history.setVersion(newConfig.getVersion());
        history.setConfigContent(newConfig.getConfigContent());
        history.setChangeReason(reason);
        history.setChangeType(changeType);
        history.setCreateBy(newConfig.getUpdateBy());
        historyMapper.insert(history);
    }
}
```

### 2.4 Controller 控制器

```java
package com.ruoyi.business.k3.controller;

import cn.dev33.satoken.annotation.SaCheckPermission;
import com.ruoyi.business.k3.domain.ErpPageConfig;
import com.ruoyi.business.k3.domain.ErpPageConfigHistory;
import com.ruoyi.business.k3.service.ErpPageConfigService;
import com.ruoyi.common.core.domain.R;
import com.ruoyi.common.web.core.BaseController;
import jakarta.validation.constraints.NotBlank;
import lombok.RequiredArgsConstructor;
import org.springframework.validation.annotation.Validated;
import org.springframework.web.bind.annotation.*;

import java.util.List;

/**
 * ERP 公共配置控制器
 */
@RequiredArgsConstructor
@RestController
@RequestMapping("/erp/config")
public class ErpPageConfigController extends BaseController {

    private final ErpPageConfigService configService;

    /**
     * 获取指定模块的配置
     */
    @GetMapping("/{moduleCode}")
    public R<String> getConfig(@PathVariable String moduleCode) {
        String config = configService.getConfigByModule(moduleCode);
        if (config == null) {
            return R.fail("未找到配置：" + moduleCode);
        }
        return R.ok(config);
    }

    /**
     * 保存配置
     */
    @SaCheckPermission("erp:config:edit")
    @PostMapping("/save")
    public R<Void> saveConfig(@Validated @RequestBody SaveConfigRequest request) {
        try {
            ErpPageConfig config = new ErpPageConfig();
            config.setConfigId(request.getConfigId());
            config.setModuleCode(request.getModuleCode());
            config.setConfigName(request.getConfigName());
            config.setConfigType(request.getConfigType());
            config.setConfigContent(request.getConfigContent());
            config.setVersion(request.getVersion());
            config.setStatus("1");
            config.setIsPublic(request.getIsPublic());
            config.setRemark(request.getRemark());

            configService.saveConfig(config, request.getChangeReason());
            return R.ok("保存成功");
        } catch (IllegalArgumentException e) {
            return R.fail("参数错误：" + e.getMessage());
        } catch (RuntimeException e) {
            return R.fail("操作失败：" + e.getMessage());
        }
    }

    /**
     * 获取配置历史列表
     */
    @SaCheckPermission("erp:config:query")
    @GetMapping("/history/{configId}")
    public R<List<ErpPageConfigHistory>> getHistory(@PathVariable Long configId) {
        List<ErpPageConfigHistory> history = configService.getConfigHistory(configId);
        return R.ok(history);
    }

    /**
     * 回滚到指定版本
     */
    @SaCheckPermission("erp:config:edit")
    @PostMapping("/rollback")
    public R<Void> rollback(@RequestBody RollbackRequest request) {
        try {
            configService.rollbackToVersion(
                request.getConfigId(), 
                request.getTargetVersion(),
                request.getReason()
            );
            return R.ok("回滚成功");
        } catch (Exception e) {
            return R.fail("回滚失败：" + e.getMessage());
        }
    }

    // ==================== 请求对象 ====================

    @lombok.Data
    public static class SaveConfigRequest {
        private Long configId;
        
        @NotBlank(message = "模块编码不能为空")
        private String moduleCode;
        
        @NotBlank(message = "配置名称不能为空")
        private String configName;
        
        @NotBlank(message = "配置类型不能为空")
        private String configType;
        
        @NotBlank(message = "配置内容不能为空")
        private String configContent;
        
        private Integer version;
        private String isPublic;
        private String remark;
        private String changeReason;
    }

    @lombok.Data
    public static class RollbackRequest {
        private Long configId;
        private Integer targetVersion;
        private String reason;
    }
}
```

```java
@Transactional(rollbackFor = Exception.class)
public void updateConfig(ErpPageConfig config) {
    ErpPageConfig old = configMapper.selectById(config.getConfigId());
```

```javascript
// utils/cache.js
const CACHE_PREFIX = 'erp:config:'
const CACHE_TTL = 10 * 60 * 1000 // 10 分钟

export function getCachedConfig(moduleCode) {
  const cached = localStorage.getItem(CACHE_PREFIX + moduleCode)
  if (cached) {
    const { data, timestamp } = JSON.parse(cached)
    if (Date.now() - timestamp < CACHE_TTL) {
      console.log('命中本地缓存:', moduleCode)
      return data
    }
    localStorage.removeItem(CACHE_PREFIX + moduleCode)
  }
  return null
}

export function cacheConfig(moduleCode, config) {
  localStorage.setItem(CACHE_PREFIX + moduleCode, JSON.stringify({
    data: config,
    timestamp: Date.now()
  }))
  console.log('缓存配置到本地:', moduleCode)
}

// Vue 组件中使用
async function loadPageConfig() {
  // 1. 尝试本地缓存
  const cached = getCachedConfig('saleOrder')
  if (cached) {
    rawConfig.value = cached
    parsedConfig.value = new ErpConfigParser(cached).parseAll()
    return
  }
  
  // 2. 调用 API
  const res = await getPageConfig('saleOrder')
  const config = typeof res.data === 'string' ? JSON.parse(res.data) : res.data
  
  // 3. 缓存到本地
  cacheConfig('saleOrder', config)
  
  // 4. 解析并渲染
  rawConfig.value = config
  parsedConfig.value = new ErpConfigParser(config).parseAll()
}
```

```java
@Service
public class ErpPageConfigService {
    
    /**
     * 获取配置（三级缓存）
     */
    public String getConfigByModule(String moduleCode) {
        String cacheKey = CACHE_KEY_PREFIX + moduleCode;
        
        // 1. 查询 Redis
        String cached = redisUtils.getCacheObject(cacheKey);
        if (StringUtils.isNotBlank(cached)) {
            log.debug(" 命中 Redis 缓存：{}", moduleCode);
            return cached;
        }
        
        // 2. 查询数据库
        ErpPageConfig config = configMapper.selectByModuleCode(moduleCode);
        if (config == null || !"1".equals(config.getStatus())) {
            return null;
        }
        
        // 3. 写入 Redis（1 小时过期）
        redisUtils.setCacheObject(cacheKey, config.getConfigContent(), 1, TimeUnit.HOURS);
        
        log.info(" 加载配置并缓存：{} (v{})", moduleCode, config.getVersion());
        return config.getConfigContent();
    }
    
    /**
     * 批量预加载配置（定时任务）
     */
    @Scheduled(cron = "0 0 2 * * ?") // 每天凌晨 2 点
    public void preloadConfigs() {
        List<ErpPageConfig> configs = configMapper.selectList(
            new LambdaQueryWrapper<ErpPageConfig>()
                .eq(ErpPageConfig::getStatus, "1")
        );
        
        for (ErpPageConfig config : configs) {
            String cacheKey = CACHE_KEY_PREFIX + config.getModuleCode();
            redisUtils.setCacheObject(cacheKey, config.getConfigContent(), 24, TimeUnit.HOURS);
        }
        
        log.info(" 预加载 {} 个配置到 Redis", configs.size());
    }
}
```
