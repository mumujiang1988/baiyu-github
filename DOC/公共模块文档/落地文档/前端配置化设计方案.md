```javascript
class PermissionEngine {
  constructor(permissionConfig) {
    this.config = permissionConfig;
  }

  hasPermission(actionCode, row = null) {
    const action = this.config.actions[actionCode];
    if (!action) return false;

    // 检查角色
    const userRoles = store.getState().user.roles;
    const hasRole = action.defaultRole.some(role => userRoles.includes(role));
    if (!hasRole) return false;

    // 检查条件
    if (action.condition && row) {
      return this.evaluateCondition(action.condition, row);
    }

    return true;
  }

  hasFieldPermission(field, action) {
    const fieldConfig = this.config.fields[field];
    if (!fieldConfig) return true;

    const userRoles = store.getState().user.roles;
    
    if (action === 'view') {
      return !fieldConfig.hidden.some(role => userRoles.includes(role));
    } else if (action === 'edit') {
      return fieldConfig.edit.some(role => userRoles.includes(role));
    }

    return true;
  }

  evaluateCondition(expression, context) {
    try {
      const func = new Function('context', `with(context) { return ${expression} }`);
      return !!func(context);
    } catch (error) {
      console.error('条件表达式执行失败:', error);
      return false;
    }
  }
}
```

```javascript
class ApprovalEngine {
  constructor(approvalConfig) {
    this.config = approvalConfig;
    this.workflow = approvalConfig.workflow || [];
  }

  getCurrentStep(row) {
    // 查找当前步骤
    return this.workflow.find(step => 
      this.evaluateCondition(step.condition, row)
    );
  }

  canAudit(row, action) {
    const currentStep = this.getCurrentStep(row);
    if (!currentStep) return false;

    // 检查角色
    const userRoles = store.getState().user.roles;
    const hasRole = userRoles.includes(currentStep.role);
    
    return hasRole && currentStep.action === action;
  }

  async execute(row, action, comment) {
    const currentStep = this.getCurrentStep(row);
    if (!currentStep) {
      throw new Error('未找到审批步骤');
    }

    // 调用后端 API 执行审批
    const response = await request({
      url: '/k3/approval/execute',
      method: 'post',
      data: {
        billId: row.id,
        step: currentStep.step,
        action: action,
        comment: comment
      }
    });

    return response;
  }

  getStatusText(status) {
    return this.config.statusMap[status] || status;
  }
}
```

**下推引擎核心代码**:
```javascript
class PushEngine {
  constructor(pushConfig) {
    this.config = pushConfig;
    this.targets = pushConfig.targets || [];
  }

  getTargets() {
    return this.targets.filter(target => target.enabled !== false);
  }

  async execute(sourceData, targetModule, confirmData = {}) {
    // 1. 查找目标配置
    const targetConfig = this.targets.find(
      t => t.targetModule === targetModule
    );
    if (!targetConfig) {
      throw new Error('未找到下推目标配置');
    }

    // 2. 字段映射
    const mappedData = this.mapFields(sourceData, targetConfig.mapping);

    // 3. 数据转换
    const transformedData = this.transformData(
      mappedData,
      targetConfig.mapping.transformation
    );

    // 4. 应用默认值
    const withDefaults = this.applyDefaults(
      transformedData,
      targetConfig.mapping.defaultValue
    );

    // 5. 合并确认数据
    const finalData = { ...withDefaults, ...confirmData };

    // 6. 数据校验
    const validation = await this.validate(
      finalData,
      targetConfig.validation.rules
    );
    if (!validation.valid) {
      throw new Error(validation.message);
    }

    // 7. 调用 API
    const result = await request({
      url: targetConfig.targetApi,
      method: 'post',
      data: finalData
    });

    // 8. 更新源单状态
    if (result.success && targetConfig.linkage) {
      await this.updateSourceStatus(sourceData, result.data);
    }

    return result;
  }

  mapFields(source, mapping) {
    const target = {};
    
    // 主表映射
    for (const [sourceField, targetField] of Object.entries(mapping.sourceToTarget)) {
      target[targetField] = source[sourceField];
    }
    
    // 明细映射
    if (source.entryList && mapping.entryMapping) {
      target.entryList = source.entryList.map(entry => {
        const mappedEntry = {};
        for (const [sourceField, targetField] of Object.entries(mapping.entryMapping)) {
          mappedEntry[targetField] = entry[sourceField];
        }
        return mappedEntry;
      });
    }
    
    return target;
  }

  applyDefaults(data, defaults) {
    const result = { ...data };
    
    for (const [field, defaultValue] of Object.entries(defaults)) {
      if (result[field] === undefined) {
        if (typeof defaultValue === 'string' && defaultValue.startsWith('${')) {
          const varName = defaultValue.slice(2, -1);
          result[field] = this.getVariable(varName);
        } else {
          result[field] = defaultValue;
        }
      }
    }
    
    return result;
  }

  getVariable(varName) {
    switch (varName) {
      case 'currentUser':
        return store.getState().user.id;
      case 'now':
        return dayjs().format('YYYY-MM-DD HH:mm:ss');
      default:
        return varName;
    }
  }
}
```

**计算引擎核心代码**:
```javascript
class CalculationEngine {
  constructor(virtualFieldConfig) {
    this.config = virtualFieldConfig;
    this.formulaCache = new Map();
  }

  calculate(field, formData) {
    const fieldConfig = this.config.fields.find(f => f.field === field);
    if (!fieldConfig) return null;

    try {
      const result = this.evaluateFormula(fieldConfig.formula, formData);
      return result;
    } catch (error) {
      console.error(`字段 ${field} 计算失败:`, error);
      return null;
    }
  }

  evaluateFormula(formula, context) {
    // 缓存已编译的公式
    if (!this.formulaCache.has(formula)) {
      const compiled = this.compileFormula(formula);
      this.formulaCache.set(formula, compiled);
    }

    const compiledFormula = this.formulaCache.get(formula);
    return compiledFormula(context);
  }

  compileFormula(formula) {
    // 处理特殊函数
    formula = formula.replace(/SUM\(([^,]+),\s*([^)]+)\)/g, (match, arrayField, field) => {
      return `this.sumArray(${arrayField}, '${field}')`;
    });

    formula = formula.replace(/AVG\(([^,]+),\s*([^)]+)\)/g, (match, arrayField, field) => {
      return `this.avgArray(${arrayField}, '${field}')`;
    });

    // 创建安全的计算函数
    return new Function('context', `
      with(context) {
        function sumArray(arr, field) {
          return arr.reduce((sum, item) => sum + (item[field] || 0), 0);
        }
        function avgArray(arr, field) {
          return arr.length > 0 ? sumArray(arr, field) / arr.length : 0;
        }
        return ${formula};
      }
    `);
  }

  sumArray(arr, field) {
    return arr.reduce((sum, item) => sum + (item[field] || 0), 0);
  }

  avgArray(arr, field) {
    return arr.length > 0 ? this.sumArray(arr, field) / arr.length : 0;
  }
}
```

```java
@Service
public class NumberRuleService {
    
    public String generateNumber(String moduleCode, NumberRuleConfig config) {
        if ("manual".equals(config.getRuleType())) {
            return null; // 手动输入
        }

        StringBuilder number = new StringBuilder();
        
        for (Segment segment : config.getSegments()) {
            switch (segment.getType()) {
                case "prefix":
                    number.append(segment.getValue());
                    break;
                case "date":
                    number.append(formatDate(segment.getFormat()));
                    break;
                case "sequence":
                    number.append(generateSequence(moduleCode, segment));
                    break;
            }
        }
        
        return number.toString();
    }
    
    private String generateSequence(String moduleCode, Segment segment) {
        // 从 Redis 获取序列号
        String key = "number:rule:" + moduleCode + ":" + segment.getReset();
        Long sequence = redisUtils.increment(key);
        
        // 格式化序列号（补零）
        return String.format("%0" + segment.getLength() + "d", sequence);
    }
```

```java
@Service
@Slf4j
@RequiredArgsConstructor
public class ErpPageConfigService {

    private final ErpPageConfigMapper configMapper;
    private final ErpPageConfigHistoryMapper historyMapper;
    @Resource
    private RedisUtils redisUtils;

    private static final String CACHE_KEY_PREFIX = "erp:page:config:";
    private static final long CACHE_TTL_HOURS = 1;

    /**
     * 根据模块编码获取配置（带多级缓存）
     */
    public String getConfigByModule(String moduleCode) {
        // 1. 尝试 Redis 缓存
        String cacheKey = CACHE_KEY_PREFIX + moduleCode;
        String cached = redisUtils.getCacheObject(cacheKey);
        if (ObjectUtil.isNotEmpty(cached)) {
            log.debug("命中 Redis 缓存：{}", moduleCode);
            return cached;
        }

        // 2. 查询数据库
        ErpPageConfig config = configMapper.selectByModuleCode(moduleCode);
        if (config == null || !"1".equals(config.getStatus())) {
            log.warn("未找到配置或配置已停用：{}", moduleCode);
            return null;
        }

        // 3. 写入 Redis 缓存
        redisUtils.setCacheObject(cacheKey, config.getConfigContent(), 
                                   CACHE_TTL_HOURS, TimeUnit.HOURS);
        
        log.info("加载配置成功：{} (v{})", moduleCode, config.getVersion());
        return config.getConfigContent();
    }

    /**
     * 保存配置（带版本管理和事务）
     */
    @Transactional(rollbackFor = Exception.class)
    public void saveConfig(ErpPageConfig config, String changeReason) {
        // 1. 验证 JSON 格式
        try {
            JSON.parseObject(config.getConfigContent());
        } catch (Exception e) {
            throw new IllegalArgumentException("配置必须是有效的 JSON 格式：" + e.getMessage());
        }

        // 2. 新增或更新
        if (config.getConfigId() != null) {
            // 更新：递增版本号
            ErpPageConfig oldConfig = configMapper.selectById(config.getConfigId());
            if (oldConfig == null) {
                throw new RuntimeException("配置不存在");
            }

            // 乐观锁检查
            if (!ObjectUtil.equal(oldConfig.getVersion(), config.getVersion())) {
                throw new RuntimeException("配置已被修改，请刷新后重试");
            }

            // 递增版本号
            config.setVersion(oldConfig.getVersion() + 1);
            configMapper.updateById(config);

            // 记录历史
            saveHistory(oldConfig, config, changeReason, "UPDATE");

            // 清除缓存
            clearCache(oldConfig.getModuleCode());

            log.info("更新配置成功：{} v{} -> v{}", 
                     oldConfig.getModuleCode(), oldConfig.getVersion(), config.getVersion());

        } else {
            // 新增：检查是否已存在
            ErpPageConfig existing = configMapper.selectByModuleCode(
                config.getModuleCode()
            );
            if (existing != null) {
                throw new RuntimeException("配置已存在：" + config.getModuleCode());
            }

            config.setVersion(1);
            configMapper.insert(config);

            // 记录历史
            saveHistory(null, config, changeReason, "ADD");

            log.info("新增配置成功：{}", config.getModuleCode());
        }
    }

    /**
     * 回滚到指定版本
     */
    @Transactional(rollbackFor = Exception.class)
    public void rollbackToVersion(Long configId, Integer targetVersion, String reason) {
        // 1. 查找目标版本历史
        ErpPageConfigHistory history = historyMapper.selectOne(
            new LambdaQueryWrapper<ErpPageConfigHistory>()
                .eq(ErpPageConfigHistory::getConfigId, configId)
                .eq(ErpPageConfigHistory::getVersion, targetVersion)
        );

        if (history == null) {
            throw new RuntimeException("指定的版本不存在：" + targetVersion);
        }

        // 2. 更新当前配置
        ErpPageConfig current = configMapper.selectById(configId);
        current.setConfigContent(history.getConfigContent());
        current.setVersion(current.getVersion() + 1);
        configMapper.updateById(current);

        // 3. 记录回滚历史
        ErpPageConfigHistory rollbackHistory = new ErpPageConfigHistory();
        rollbackHistory.setConfigId(configId);
        rollbackHistory.setModuleCode(current.getModuleCode());
        rollbackHistory.setConfigType(current.getConfigType());
        rollbackHistory.setVersion(current.getVersion());
        rollbackHistory.setConfigContent(history.getConfigContent());
        rollbackHistory.setChangeReason(reason);
        rollbackHistory.setChangeType("ROLLBACK");
        historyMapper.insert(rollbackHistory);

        // 4. 清除缓存
        clearCache(current.getModuleCode());

        log.info("回滚配置成功：{} v{} -> rollback to v{}", 
                 current.getModuleCode(), current.getVersion(), targetVersion);
    }

    /**
     * 清除缓存
     */
    private void clearCache(String moduleCode) {
        String cacheKey = CACHE_KEY_PREFIX + moduleCode;
        redisUtils.deleteObject(cacheKey);
        log.debug("清除缓存：{}", cacheKey);
    }

    /**
     * 保存历史记录
     */
    private void saveHistory(ErpPageConfig oldConfig, ErpPageConfig newConfig, 
                            String reason, String changeType) {
        ErpPageConfigHistory history = new ErpPageConfigHistory();
        history.setConfigId(newConfig.getConfigId());
        history.setModuleCode(newConfig.getModuleCode());
        history.setConfigType(newConfig.getConfigType());
        history.setVersion(newConfig.getVersion());
        history.setConfigContent(newConfig.getConfigContent());
        history.setChangeReason(reason);
        history.setChangeType(changeType);
        history.setCreateBy(newConfig.getUpdateBy());
        historyMapper.insert(history);
    }
}
```

```vue
    const parser = new ErpConfigParser(rawConfig.value)
    parsedConfig.value = parser.parseAll()
    
    console.log('配置加载成功:', rawConfig.value)
    
  } catch (error) {
    console.error('配置加载失败:', error)
    configError.value = `加载配置失败：${error.message}`
    
  } finally {
    configLoading.value = false
  }
}

// ==================== 导出 ====================
defineExpose({
  refreshConfig
})
</script>
```

## ❓ 常见问题

### Q1: 配置文件过大怎么办？

**A**: 采用模块化拆分：
```json
// 拆分为多个文件
{
  "search.config.json": {...},
  "table.config.json": {...},
  "form.config.json": {...},
  "actions.config.json": {...},
  "permissions.config.json": {...}
}
```

### Q2: 如何保证配置的安全性？

**A**: 
1. JSON Schema 验证配置格式
2. 表达式沙箱环境
3. 后端二次权限校验
4. 敏感数据加密传输

### Q3: 性能如何优化？

**A**: 
1. 配置分块加载
2. 字典数据缓存（Redis）
3. 公式计算优化（Web Worker）
4. 分页加载、虚拟滚动
5. 多级缓存架构（Local + Redis + DB）

### Q4: 如何处理复杂的业务流程？

**A**: 
1. 使用审批流程配置
2. 结合下推机制
3. 添加业务规则引擎
4. 必要时编写自定义代码
