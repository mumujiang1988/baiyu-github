# ========== 阶段 1: 清理准备 ==========
# 1.1 强制终止所有 Java 进程（避免端口占用、进程冲突）
Write-Host "`n========================================" -ForegroundColor Cyan
Write-Host "   RuoYi-WMS 快速启动脚本 (ERP 开发优化版)" -ForegroundColor Green
Write-Host "========================================`n" -ForegroundColor Cyan

Write-Host "=== [1/4] 终止 Java 进程 ===" -ForegroundColor Yellow
$javaCount = (Get-Process | Where-Object {$_.ProcessName -like "java*"}).Count
if ($javaCount -gt 0) {
    Write-Host "  发现 $javaCount 个 Java 进程，正在终止..." -ForegroundColor Gray
    taskkill /f /im java.exe 2>$null
    taskkill /f /im javaw.exe 2>$null
    Start-Sleep -Seconds 2
    Write-Host "  √ Java 进程已终止" -ForegroundColor Green
} else {
    Write-Host "  ℹ 无需终止进程" -ForegroundColor Gray
}

# 1.2 切换到项目根目录
Write-Host "`n=== [2/4] 清理模块缓存 ===" -ForegroundColor Cyan
$projectRoot = "D:\baiyuyunma\baiyu-github\baiyu-github\baiyu-ruoyi"
Set-Location $projectRoot

# 1.3 智能清理所有相关模块的 target 目录（按编译顺序清理）
$targetPaths = @(
    ".\ruoyi-admin-wms\target",
    ".\ruoyi-modules\ruoyi-erp-api\target",
    ".\ruoyi-modules\ruoyi-system\target"
)

$cleanedCount = 0
foreach ($path in $targetPaths) {
    if (Test-Path $path) {
        try {
            Remove-Item -Recurse -Force $path -ErrorAction Stop
            Write-Host "  √ 已清理：$path" -ForegroundColor Green
            $cleanedCount++
        } catch {
            Write-Host "  ⚠ 清理失败：$path (Maven 将自动清理)" -ForegroundColor Yellow
        }
    } else {
        Write-Host "  - 跳过 (不存在): $path" -ForegroundColor Gray
    }
}
Write-Host "共清理 $cleanedCount/$($targetPaths.Count) 个核心模块目录" -ForegroundColor Cyan

# ========== 阶段 2: Maven 编译（最小可用模块集） ==========
Write-Host "`n=== [3/4] Maven 编译核心模块 ===" -ForegroundColor Cyan
Write-Host "  编译范围：ruoyi-admin-wms + ruoyi-erp-api + ruoyi-system + 依赖" -ForegroundColor Gray
Write-Host "  编译参数：-T 8 (8 线程并行) | -DskipTests (跳过测试)`n" -ForegroundColor Gray

$stopwatch = [System.Diagnostics.Stopwatch]::StartNew()
mvn clean install -pl ruoyi-admin-wms,ruoyi-modules/ruoyi-erp-api,ruoyi-modules/ruoyi-system -am -DskipTests -T 8
$stopwatch.Stop()

if ($LASTEXITCODE -ne 0) {
    Write-Host "`n⚠ Maven 编译失败！请检查上方错误信息" -ForegroundColor Red
    Write-Host "`n建议操作:" -ForegroundColor Yellow
    Write-Host "  1. 检查代码是否有语法错误" -ForegroundColor Gray
    Write-Host "  2. 确认依赖是否都已提交" -ForegroundColor Gray
    Write-Host "  3. 尝试完整编译：mvn clean install -DskipTests`n" -ForegroundColor Gray
    exit 1
}

Write-Host "`n√ 编译成功！耗时：$([math]::Round($stopwatch.Elapsed.TotalSeconds, 1)) 秒" -ForegroundColor Green

# ========== 阶段 3: 启动后端服务 ==========
Write-Host "`n=== [4/4] 启动后端服务 ===" -ForegroundColor Cyan
$adminPath = "D:\baiyuyunma\baiyu-github\baiyu-github\baiyu-ruoyi\ruoyi-admin-wms"
Set-Location $adminPath

Write-Host "提示:" -ForegroundColor Yellow

Write-Host "  • 前端启动：cd ..\..\baiyu-web; npm run dev" -ForegroundColor Gray
Write-Host "  • 查看日志：tail -f logs/sys-info.log" -ForegroundColor Gray
Write-Host "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━`n" -ForegroundColor Cyan
