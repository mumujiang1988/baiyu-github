# ========== 阶段 1: 清理准备 ==========
# 1.1 强制终止所有 Java 进程（避免端口占用、进程冲突）
Write-Host "=== 终止 Java 进程 ===" -ForegroundColor Yellow
taskkill /f /im java.exe 2>$null
taskkill /f /im javaw.exe 2>$null
Start-Sleep -Seconds 2  # 等待 2 秒确保进程完全终止，文件释放

# 1.2 切换到项目根目录
Write-Host "=== 切换到项目目录 ===" -ForegroundColor Cyan
$projectRoot = "D:\baiyuyunma\baiyu-github\baiyu-github\baiyu-ruoyi"
Set-Location $projectRoot

# 1.3 智能清理 target 目录（仅清理存在的目录，避免报错）
Write-Host "=== 清理 target 目录 ===" -ForegroundColor Cyan
$targetPaths = @(
    ".\ruoyi-modules\ruoyi-system\target",
    ".\ruoyi-modules\ruoyi-demo\target",
    ".\ruoyi-modules\ruoyi-erp-api\target",
    ".\ruoyi-modules\ruoyi-generator\target",
    ".\ruoyi-admin-wms\target",
    ".\ruoyi-common\ruoyi-common-mybatis\target"  # 添加之前失败的模块
)

$successCount = 0
foreach ($path in $targetPaths) {
    if (Test-Path $path) {
        try {
            Remove-Item -Recurse -Force $path -ErrorAction Stop
            Write-Host "  √ 已清理：$path" -ForegroundColor Green
            $successCount++
        } catch {
            Write-Host "  ⚠ 清理失败：$path (可能被占用，Maven 将自动清理)" -ForegroundColor Yellow
        }
    } else {
        Write-Host "  - 跳过 (不存在): $path" -ForegroundColor Gray
    }
}
Write-Host "共清理 $successCount/$($targetPaths.Count) 个目录" -ForegroundColor Cyan

# ========== 阶段 2: Maven 编译 ==========
Write-Host "`n=== 开始 Maven 多线程编译 ===" -ForegroundColor Green
mvn clean install -DskipTests -T 16
if ($LASTEXITCODE -ne 0) {
    Write-Host "`n⚠ Maven 编译失败，请检查错误信息" -ForegroundColor Red
    exit 1
}
Write-Host "`n√ Maven 编译成功！`n" -ForegroundColor Green

# ========== 阶段 3: 启动后端服务 ==========
Write-Host "`n=== 启动后端服务 (ruoyi-admin-wms) ===" -ForegroundColor Cyan
$adminPath = "D:\baiyuyunma\baiyu-github\baiyu-github\baiyu-ruoyi\ruoyi-admin-wms"
Set-Location $adminPath
mvn spring-boot:run

# ========== 阶段 4: 启动前端服务 (新开窗口) ==========
# 注意：前端启动命令在新窗口执行，避免阻塞当前窗口
Write-Host "`n=== 启动前端服务 (新窗口) ===" -ForegroundColor Cyan
Start-Process powershell -ArgumentList "Set-Location 'D:\baiyuyunma\baiyu-github\baiyu-github\baiyu-web'; npm run dev"
