### **方案 C: 添加字段转换层（最佳实践 ⭐⭐⭐⭐⭐）**

在前端 API 调用时添加转换层:

#### **创建工具函数**

**文件**: `baiyu-web/src/views/erp/utils/salespersonHelper.js`

```javascript
/**
 * 销售人员字典辅助工具
 */

/**
 * 格式化销售人员选项为查询条件
 * @param {string|number} selectedValue - 选中的值
 * @param {Array} dictOptions - 字典选项
 * @returns {Object} 查询条件
 */
export function formatSalespersonCondition(selectedValue, dictOptions) {
  if (!selectedValue) {
    return null
  }
  
  // 查找对应的销售人员
  const salesperson = dictOptions.find(opt => {
    // 支持多种匹配方式
    return opt.fseller === String(selectedValue) || 
           opt.value === selectedValue ||
           opt.FSalerId === String(selectedValue)
  })
  
  if (!salesperson) {
    console.warn('未找到对应的销售人员:', selectedValue)
    return null
  }
  
  // 返回标准查询条件
  return {
    field: 'fseller',           // 统一使用 fseller
    operator: 'eq',
    value: salesperson.fseller  // 使用 salesman_id
  }
}

/**
 * 获取销售人员的显示名称
 */
export function getSalespersonLabel(value, dictOptions) {
  const salesperson = dictOptions.find(
    opt => opt.fseller === String(value) || opt.value === value
  )
  return salesperson ? salesperson.label : ''
}
```

**使用示例**:
```javascript
import { formatSalespersonCondition } from '@/views/erp/utils/salespersonHelper'

// 在查询条件构建中使用
const queryConditions = () => {
  const salespersonCond = formatSalespersonCondition(
    formData.fseller,
    dictionaryManager.getDict('salespersons')
  )
  
  return {
    conditions: salespersonCond ? [salespersonCond] : []
  }
}
```
