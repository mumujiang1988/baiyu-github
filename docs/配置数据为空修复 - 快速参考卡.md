## ✅ **修复要点**

### **1. 正确解析 ErpResponse**

```javascript
// ❌ 修复前
const data = res.data || res

// ✅ 修复后
let data
if (res.code === 200 || res.code === 0) {
  data = res.data || {}  // 从 res.data 提取
} else {
  throw new Error(res.msg || '加载失败')
}

// ✅ 检查必要字段
if (!data.configId) {
  throw new Error('配置不存在或已删除')
}
```

---

### **2. 处理空配置内容**

```javascript
// ❌ 修复前
let configContent = data.configContent || ''
JSON.parse(configContent)  // ❌ 空字符串会崩溃

// ✅ 修复后
let configContent = data.configContent || ''

// 检测并初始化空字符串
if (!configContent || configContent.trim() === '') {
  console.warn('⚠️ 配置内容为空，将使用空对象初始化')
  configContent = '{}'
}

try {
  const parsed = JSON.parse(configContent)
  configContent = JSON.stringify(parsed, null, 2)
} catch (e) {
  console.error('❌ JSON 解析失败:', e.message)
  // 保持原样，让用户手动处理
}
```

### **4. 优化调用方法**

#### **handleEdit (编辑)**
```javascript
// ✅ 修复后
function handleEdit(row) {
  console.log('\n✏️ [配置管理] 点击编辑，configId:', row.configId)
  
  isEditMode.value = false  // 先重置状态
  currentConfig.value = { ...row }
  
  // 加载数据并打开对话框
  loadEditData(row.configId)
    .then(() => {
      console.log('✅ 数据加载成功，打开编辑对话框')
      viewDialogVisible.value = true
    })
    .catch(error => {
      console.error('❌ 数据加载失败，不打开对话框')
      // 错误已在 loadEditData 中处理
    })
}
```

#### **handleView (查看)**
```javascript
// ✅ 修复后
function handleView(row) {
  console.log('\n👁️ [配置管理] 点击查看，configId:', row.configId)
  
  currentConfig.value = { ...row }
  isEditMode.value = false
  
  // 加载数据
  loadEditData(row.configId)
    .then(() => {
      console.log('✅ 数据加载成功，打开查看对话框')
      viewDialogVisible.value = true
    })
    .catch(error => {
      console.error('❌ 数据加载失败')
      // 即使失败也打开对话框（显示错误信息）
      viewDialogVisible.value = true
    })
}
```
