### 步骤 5：在表格列中使用字典

```vue
<template>
  <el-table :data="tableData">
    <!-- 币别列（使用字典渲染） -->
    <el-table-column prop="currency" label="币别">
      <template #default="{ row }">
        {{ getCurrencyLabel(row.currency) }}
      </template>
    </el-table-column>
    
    <!-- 订单状态列（标签渲染） -->
    <el-table-column prop="orderStatus" label="订单状态">
      <template #default="{ row }">
        <el-tag :type="getStatusType(row.orderStatus)">
          {{ getStatusLabel(row.orderStatus) }}
        </el-tag>
      </template>
    </el-table-column>
  </el-table>
</template>

<script setup>
import dictionaryManager from '@/views/erp/utils/DictionaryManager'

// 获取币别标签
const getCurrencyLabel = (value) => {
  const dict = dictionaryManager.getDict('currency')
  const item = dict.find(d => d.value === value)
  return item?.label || value
}

// 获取状态标签
const getStatusLabel = (value) => {
  const dict = dictionaryManager.getDict('orderStatus')
  const item = dict.find(d => d.value === value)
  return item?.label || value
}

// 获取标签类型
const getStatusType = (value) => {
  const dict = dictionaryManager.getDict('orderStatus')
  const item = dict.find(d => d.value === value)
  return item?.type || 'info'
}
</script>
```

### 场景 2：远程搜索字典（国家）

```vue
<template>
  <el-select
    v-model="form.country"
    placeholder="请选择国家"
    filterable
    remote
    :remote-method="searchCountry"
    :loading="loading"
  >
    <el-option
      v-for="item in countryOptions"
      :key="item.id"
      :label="`${item.labelZh} (${item.labelEn})`"
      :value="item.id"
    />
  </el-select>
</template>

<script setup>
import { ref, onMounted } from 'vue'
import request from '@/utils/request'

const form = ref({})
const countryOptions = ref([])
const loading = ref(false)

// 搜索国家
const searchCountry = async (keyword) => {
  if (!keyword) return
  
  loading.value = true
  try {
    const response = await request({
      url: '/erp/engine/country/search',
      method: 'get',
      params: { keyword, limit: 20 }
    })
    
    countryOptions.value = response.data || []
  } finally {
    loading.value = false
  }
}

// 初始化时加载常用国家
onMounted(async () => {
  const response = await request({
    url: '/erp/engine/country/list',
    method: 'get'
  })
  countryOptions.value = response.data || []
})
</script>
```

## ⚡ 性能提示

### 1. 避免重复加载

```javascript
// ❌ 错误：每个组件都单独加载
onMounted(async () => {
  await dictionaryManager.loadAll()
})

// ✅ 正确：只在主入口加载一次
// DictionaryManager 会自动缓存，多次调用不会重复请求
```

### 2. 使用计算属性

```javascript
// ❌ 错误：每次渲染都调用 getDict
const getOptions = () => dictionaryManager.getDict('currency')

// ✅ 正确：使用计算属性缓存结果
const currencyOptions = computed(() => 
  dictionaryManager.getDict('currency')
)
```

### 3. 批量获取字典

```javascript
// ✅ 推荐：一次性获取多个字典
const dicts = {
  currency: dictionaryManager.getDict('currency'),
  salespersons: dictionaryManager.getDict('salespersons'),
  paymentTerms: dictionaryManager.getDict('paymentTerms')
}
```

## 🐛 常见问题

### Q1: 字典数据为空？

```javascript
// 检查是否已加载
const status = dictionaryManager.getStatus()
console.log(status.loaded) // true/false

// 强制重新加载
await dictionaryManager.loadAll(true)
```

### Q2: 找不到指定的字典类型？

```javascript
// 查看所有可用的字典类型
console.log(dictionaryManager.getStatus().dictTypes)

// 检查拼写是否正确
const dict = dictionaryManager.getDict('currency') // 注意大小写
```

### Q3: 如何清除缓存？

```javascript
// 清除单个字典（暂不支持）
// dictionaryManager.clear('currency') // ❌

// 清除所有缓存
dictionaryManager.clear() // ✅

// 重新加载
await dictionaryManager.loadAll(true)
```
